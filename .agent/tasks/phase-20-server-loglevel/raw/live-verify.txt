[gating] no wire call before approval + confirmation
  set_log_level, approval unset           -> status=disabled
  set_log_level, bad confirmation token   -> status=gap
  set_log_level, unknown level name       -> status=error (valid_levels returned)

[set_log_level] SetLogLevel (per-node level, readback)
  GET     node_log_level={'Server': 'LOG_LEVEL_INFO'}
  SET     -> LOG_LEVEL_DEBUG  -> status=applied node_log_level={'Server': 'LOG_LEVEL_DEBUG'} failed=[]
  RESTORE -> LOG_LEVEL_INFO   -> readback={'Server': 'LOG_LEVEL_INFO'} (matches original)

[drop_logs] DropLogs (authorized irreversible delete of server log history)
  DROP    nodes=[] (current node) -> status=applied failed_nodes=[]
  POST    GetLogLevel={'Server': 'LOG_LEVEL_INFO'} (server healthy after drop)
