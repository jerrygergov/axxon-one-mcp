[set_timezone] SetTimeZone (OS timezone, optional DST-off, readback)
  GET     current=Arabian Standard Time daylight_saving_mode_off=False
  SET     -> UTC (daylight_saving_mode_off=True)  -> status=applied current=UTC
  RESTORE -> Arabian Standard Time                -> readback=Arabian Standard Time

[set_ntp] SetNTP (ntp_url, sync_ip_devices, optional refresh Duration)
  GET     ntp_url='' (no NTP configured)
  SET     -> pool.ntp.org sync_ip_devices=True refresh_rate_s=3600 -> status=applied
  READBACK-> pool.ntp.org / True / 3600s
  RESTORE -> ntp_url='' (cleared back to original empty state)
